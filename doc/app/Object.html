def get_page_body(url)

def get_page_keyword(url, keyword)
  agent = Mechanize.new
  page = agent.get(url)
  NKF.nkf('-wm0', page.body).to_s.include?(keyword) ? keyword : "There is no keyword '#{keyword}'. #{Time.now}"
end

def get_page_title(url)
  agent = Mechanize.new
  page = agent.get(url)
  NKF.nkf('-wm0', page.title) || 'no title'
end
